----
$ brew install go
$ cat >> .bashrc << 'EOF'
$ export GOPATH="${HOME}/go"
$ export GOROOT="$(brew --prefix golang)/libexec"
$ export PATH="$PATH:${GOPATH}/bin:${GOROOT}/bin"
$ EOF
$ source .bashrc
$ go get golang.org/x/tour/gotour
$ gotour
----
